139  size_t extended_message(char *buffer, size_t len, unsigned &code1, unsigned &code2) const noexcept
140  {

157  size_t raw_backtrace(void **buffer, size_t len) const noexcept
158  {
159  auto &log = extended_error_code_log();
160  if (!log.valid(unique_id(_unique_id)))
161  return 0;
162  auto &item = log[unique_id(_unique_id)];
163  size_t written = 0;
164  void ** _backtrace = (void **)item.backtrace;
165  for (size_t n = 0; n < len && n < sizeof(item.backtrace) / sizeof(item.backtrace[0]); n++)
166  {
167  buffer[n] = _backtrace[n];
168  written++;
169  }
170  return written;
171  }
173  char **backtrace() const noexcept
174  {
175  auto &log = extended_error_code_log();
176  if (!log.valid(unique_id(_unique_id)))
177  return 0;
178  auto &item = log[unique_id(_unique_id)];
179  char **ret = nullptr;
180  if (!item.using_backtrace)
181  {
182  ret = (char **)calloc(2 * sizeof(char *) + sizeof(item.function) + 1, 1);
183  if (!ret)
184  return nullptr;
185  ret[0] = (char *)ret + 2 * sizeof(char *);
186  strncpy(ret[0], item.function, sizeof(item.function));
187  return ret;
188  }
189  else
190  ret = backtrace_symbols((void **)item.backtrace, sizeof(item.backtrace) / sizeof(item.backtrace[0]));
191  if (!log.valid(unique_id(_unique_id)))
